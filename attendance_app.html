Google Apps Script — Copy & Deploy

Follow these steps to connect Google Sheets for free:

  1. Open Google Sheets → create a new sheet named AttendanceDB
  2. Go to Extensions → Apps Script
  3. Delete existing code, paste the script below, click Save (💾)
  4. Click Deploy → New Deployment → Web App
  5. Set "Execute as" = Me, "Who has access" = Anyone → Deploy
  6. Copy the Web App URL and paste it above in this admin panel
function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const type = data.type; if (type === "punch") { let sheet = ss.getSheetByName("Attendance"); if (!sheet) sheet = ss.insertSheet("Attendance"); if (sheet.getLastRow() === 0) { sheet.appendRow(["Date","EmpID","Name","CheckIn","BreakOut","BreakIn","CheckOut","WorkingHrs","BreakHrs","Device","Status"]); } const rows = sheet.getDataRange().getValues(); const today = data.date; const empId = data.empId; let rowIndex = -1; for (let i = 1; i < rows.length; i++) { if (rows[i][0] === today && rows[i][1] === empId) { rowIndex = i + 1; break; } } if (rowIndex === -1) { sheet.appendRow([today, empId, data.name, data.checkIn||"", data.breakOut||"", data.breakIn||"", data.checkOut||"", data.workingHrs||"", data.breakHrs||"", data.device, data.status]); } else { const cols = {checkIn:4,breakOut:5,breakIn:6,checkOut:7,workingHrs:8,breakHrs:9,device:10,status:11}; for (const [k,c] of Object.entries(cols)) { if (data[k] !== undefined) sheet.getRange(rowIndex, c).setValue(data[k]); } } } if (type === "register") { let sheet = ss.getSheetByName("Employees"); if (!sheet) sheet = ss.insertSheet("Employees"); if (sheet.getLastRow() === 0) { sheet.appendRow(["EmpID","Name","Password","DeviceID","RegisteredAt"]); } sheet.appendRow([data.empId, data.name, data.password, data.deviceId||"", new Date().toISOString()]); } if (type === "lockDevice") { let sheet = ss.getSheetByName("Employees"); if (sheet) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]).toUpperCase() === String(data.empId).toUpperCase()) { sheet.getRange(i+1, 4).setValue(data.deviceId); break; } } } } return ContentService.createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:"error", msg: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action; if (action === "getReport") { const sheet = ss.getSheetByName("Attendance"); if (!sheet) return ContentService.createTextOutput(JSON.stringify({rows:[]})).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const empId = e.parameter.empId; const month = e.parameter.month; const result = rows.slice(1) .filter(r => (!empId || r[1]===empId) && (!month || String(r[0]).startsWith(month))) .map(r => Object.fromEntries(headers.map((h,i)=>[h,r[i]]))); return ContentService.createTextOutput(JSON.stringify({rows:result})).setMimeType(ContentService.MimeType.JSON); } if (action === "verifyEmployee") { const sheet = ss.getSheetByName("Employees"); if (!sheet) return ContentService.createTextOutput(JSON.stringify({valid:false,msg:"No Employees sheet found"})).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); const empId = String(e.parameter.empId||"").toUpperCase().trim(); const password = String(e.parameter.password||"").trim(); const match = rows.slice(1).find(r => String(r[0]).toUpperCase().trim() === empId && String(r[2]).trim() === password ); if (match) { return ContentService.createTextOutput(JSON.stringify({ valid: true, name: match[1], deviceId: match[3] || "" })).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({valid:false,msg:"ID ya password galat hai"})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput("SmartAttend API OK").setMimeType(ContentService.MimeType.TEXT); }
