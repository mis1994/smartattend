function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var type = data.type; if (type === "punch") { var sheet = ss.getSheetByName("Attendance"); if (!sheet) sheet = ss.insertSheet("Attendance"); if (sheet.getLastRow() === 0) { sheet.appendRow(["Date","EmpID","Name","CheckIn","BreakOut", "BreakIn","CheckOut","WorkingHrs","BreakHrs","Distance(m)","Device","Status"]); } var rows = sheet.getDataRange().getValues(); var today = data.date; var empId = data.empId; var rowIndex = -1; for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === today && String(rows[i][1]) === empId) { rowIndex = i + 1; break; } } if (rowIndex === -1) { sheet.appendRow([today, empId, data.name, data.checkIn||"", data.breakOut||"", data.breakIn||"", data.checkOut||"", data.workingHrs||"", data.breakHrs||"", data.distance||"", data.device, data.status]); } else { var cols = {checkIn:4,breakOut:5,breakIn:6,checkOut:7, workingHrs:8,breakHrs:9,distance:10,device:11,status:12}; for (var k in cols) { if (data[k] !== undefined && data[k] !== "") sheet.getRange(rowIndex, cols[k]).setValue(data[k]); } } } if (type === "register") { var esheet = ss.getSheetByName("Employees"); if (!esheet) esheet = ss.insertSheet("Employees"); if (esheet.getLastRow() === 0) esheet.appendRow(["EmpID","Name","Password","DeviceID","RegisteredAt"]); esheet.appendRow([data.empId, data.name, data.password, data.deviceId||"", new Date().toISOString()]); } if (type === "lockDevice") { var lsheet = ss.getSheetByName("Employees"); if (lsheet) { var lrows = lsheet.getDataRange().getValues(); for (var j = 1; j < lrows.length; j++) { if (String(lrows[j][0]).toUpperCase().trim() === String(data.empId).toUpperCase().trim()) { lsheet.getRange(j+1, 4).setValue(data.deviceId); break; } } } } return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action; if (action === "verifyEmployee") { var sheet = ss.getSheetByName("Employees"); if (!sheet) return ContentService .createTextOutput(JSON.stringify({valid:false,msg:"Employees sheet nahi mili"})) .setMimeType(ContentService.MimeType.JSON); var rows = sheet.getDataRange().getValues(); var empId = String(e.parameter.empId||"").toUpperCase().trim(); var password = String(e.parameter.password||"").trim(); var match = null; for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).toUpperCase().trim() === empId && String(rows[i][2]).trim() === password) { match = rows[i]; break; } } if (match) return ContentService .createTextOutput(JSON.stringify({valid:true,name:match[1],deviceId:match[3]||""})) .setMimeType(ContentService.MimeType.JSON); return ContentService .createTextOutput(JSON.stringify({valid:false,msg:"ID ya password galat hai"})) .setMimeType(ContentService.MimeType.JSON); } if (action === "getReport") { var rsheet = ss.getSheetByName("Attendance"); if (!rsheet) return ContentService .createTextOutput(JSON.stringify({rows:[]})) .setMimeType(ContentService.MimeType.JSON); var rrows = rsheet.getDataRange().getValues(); var headers = rrows[0]; var filterEmp = e.parameter.empId; var filterMonth = e.parameter.month; var filterDate = e.parameter.date; var result = []; for (var r = 1; r < rrows.length; r++) { var row = rrows[r]; var matchEmp = !filterEmp || String(row[1]).toUpperCase()===String(filterEmp).toUpperCase(); var matchMonth = !filterMonth || String(row[0]).indexOf(filterMonth)===0; var matchDate = !filterDate || String(row[0])===filterDate; if (matchEmp && matchMonth && matchDate) { var obj = {}; for (var h = 0; h < headers.length; h++) obj[headers[h]] = row[h]; result.push(obj); } } return ContentService .createTextOutput(JSON.stringify({rows:result})) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput("SmartAttend API OK") .setMimeType(ContentService.MimeType.TEXT); }
